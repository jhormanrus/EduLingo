## Getting started
### Include the neccesary themes in main.scss:

```scss
@charset 'UTF-8';

// ng-select theme
@import '~@ng-select/ng-select/themes/default.theme.css';
// ngx-datatable theme
@import '~@swimlane/ngx-datatable/index.css';
@import '~@swimlane/ngx-datatable/themes/material.scss';
@import '~@swimlane/ngx-datatable/assets/icons.css';
// ngx-ui-switch theme
@import '~ngx-ui-switch/ui-switch.component.scss';
// global theme
@import '../../../../../libs/theme/scss/dashboard.scss';
// custom scrollbar
@import '../../../../../libs/theme/scss/scrollbar.scss';
// ng-select theme
@import '../../../../../libs/theme/libraries/ng-select.scss';
// ngx-datatable theme
@import '../../../../../libs/theme/libraries/ngx-datatable.scss';
// bootstrap-notify.js theme
@import '../../../../../libs/theme/libraries/alert.scss';
// dldatetimepicker theme
@import '../../../../../libs/theme/libraries/datetimepicker.scss';
// ngbootstrap theme
@import '../../../../../libs/theme/libraries/ng-bootstrap.scss';
// ngx-ui-switch theme
@import '../../../../../libs/theme/libraries/ngx-ui-switch.scss';
// angular-editor theme
@import '../../../../../libs/theme/libraries/angular-editor.scss';

```
